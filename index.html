Live Demo - Channel Configuration

Sample Arduino Code:

#include <WiFi.h>
#include <HTTPClient.h>

String apiKey = "YOUR_API_KEY";
const char* server = "iotanalyze.com";

void sendData(float temp, float humidity) {
  HTTPClient http;
  String url = "/update?api_key=" + apiKey;
  url += "&field1=" + String(temp);
  url += "&field2=" + String(humidity);
  http.begin(server + url);
  http.GET();
}
